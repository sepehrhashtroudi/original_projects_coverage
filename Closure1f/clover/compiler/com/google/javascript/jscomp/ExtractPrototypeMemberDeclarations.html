26    /**
27    * When there are multiple prototype member declarations to the same class,
28    * use a temp variable to alias the prototype object.
29    *
30    * Example:
31    *
32    * <pre>
33    * function B() { ... } \
34    * B.prototype.foo = function() { ... } \___ {@link ExtractionInstance}
35    * ... /
36    * B.prototype.bar = function() { ... } /
37    * ^---------------------------------{@link PrototypeMemberDeclaration}
38    * </pre>
39    * <p>becomes
40    * <pre>
41    * function B() { ... }
42    * x = B.prototype;
43    * x.foo = function() { ... }
44    * ...
45    * x.bar = function() { ... }
